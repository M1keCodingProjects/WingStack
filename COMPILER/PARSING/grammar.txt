Program    : (Expression SPACE? EOL)*
Expression : Procedure
Procedure  : PrintProc
PrintProc  : "print" SPACE StackExpr (SPACE "with" ERROR)?
StackExpr  : ((StackValue | STACKOP | CallChain) SPACE)*
StackValue : NUM | STR
CallChain  : Property (("." Property) | IDProp)?*
Property   : IDProp
FuncCall   : WORD "(" SPACE? StackExpr ")"
IDProp     : "[" SPACE? StackExpr "]"
