Program: Expression*
    Expression: Procedure | Assignment | (CallChain ".")? FuncCall
        Procedure: PrintProcedure | ...
            PrintProcedure: "print" " " StackExpr
                StackExpr: (Value | STACKOP_TOKEN | CallChain )* " "
                    Value: NUM_TOKEN | STR_TOKEN
    
    CallChain: Property("." Property)?*
        Property: WORD_TOKEN | FuncCall | IndexedProperty
            FuncCall: WORD_TOKEN "(" StackExpr ")"
            IndexedProperty: "[" StackExpr "]"
